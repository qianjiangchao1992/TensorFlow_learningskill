标签：同时满足以下4个标签
1、城市：合肥
2、排除：12个月内有广发信用卡账单用户
3、话费：Apru 50元以上
4、最近12个月，至少有一封信用卡账单
##########
#创建广发信用卡
create table if not exists temp.guangfacredit as 
select distinct mobile as mobile from dw_ods.base_a2p_creditcard_bill_info
where cust_type=2031;
#创建信用卡账单（不含广发信用卡）
create table if not exists temp.credit_all_new as 
select distinct t1.mobile as mobile from dw_ods.base_a2p_creditcard_bill_info t1
where not exists (select 1 from temp.guangfacredit t0 where t0.mobile=t1.mobile);

###广发信用卡可发卡城市(包括手机、城市）
create table if not exists temp.guangfa_allcity_info as 
select distinct t1.id as mobile,t4.city as city  from 
tags.tag_1202_arpu t1,
temp.credit_all_new t2,
tags.tag_1102_owner_city t3,
temp.guangfacredit_city t4
where t1.id=t3.id and t1.id=t2.mobile and t1.dated='2018-07' and t3.dated='2018-07'
and t1.tag_value>50 and t3.tag_value=t4.city ;


######选取合肥的######
select count(*) from temp.guangfa_allcity_info where city='安徽合肥';
